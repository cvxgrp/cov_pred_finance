28def _cholesky_precision(cov): 

29 """ 

30 Computes Cholesky factor of the inverse of each covariance matrix in cov

72class _CombinationProblem: 

73 def __init__(self, keys, n, window): 

74 self.keys = keys 

75 K = len(keys) 

76 self._weight = cvx.Variable(len(keys), name="weights") 

77 self.A_param = cvx.Parameter((n * window, K)) 

78 self.P_chol_param = cvx.Parameter((K, K)) 

79 

80 @property 

81 def _constraints(self): 

82 return [cvx.sum(self._weight) == 1, self._weight >= 0] 

83 

84 @property 

85 def _objective(self): 

86 return cvx.sum(cvx.log(self.A_param @ self._weight)) - 0.5 * cvx.sum_squares(self.P_chol_param.T @ self._weight) 

87 

88 def _construct_problem(self): 

89 self.prob = cvx.Problem(cvx.Maximize(self._objective), self._constraints) 

90 

91 def solve(self, **kwargs): 

92 return self.prob.solve(**kwargs) 

93 

94 @property 

95 def weights(self): 

96 return pd.Series(index=self.keys, data=self._weight.value) 

97 

98 @property 

99 def status(self): 

100 return self.prob.status

103def from_ewmas(returns, pairs, min_periods_vola=20, min_periods_cov=20, clip_at=None, mean=False): 

104 """ 

105 Estimate a series of covariance matrices using the iterated EWMA method 

106 

107 param returns: Frame of returns 

108 param pairs: list of pairs of EWMA half lives, e.g. [(20, 20), (10, 50), (20, 60)], 

109 pair[0] is the half life for volatility estimation 

110 pair[1] is the half life for covariance estimation 

111 param min_periods_vola: minimum number of observations to start EWMA for volatility estimation (optional) 

112 param min_periods_cov: minimum number of observations to start EWMA for covariance estimation (optional) 

113 param clip_at: clip volatility adjusted returns at +- clip_at (optional) 

114 param mean: subtract EWMA mean from returns and volatility adjusted returns (optional) 

115 

116 return: Yields tuples with time, mean, covariance matrix, weights 

117 """ 

118 # compute the covariance matrices, one time series for each pair 

119 results = { 

120 f"{pair[0]}-{pair[1]}": { 

121 result.time: result 

122 for result in iterated_ewma( 

123 returns=returns, 

124 vola_halflife=pair[0], 

125 cov_halflife=pair[1], 

126 min_periods_vola=min_periods_vola, 

127 min_periods_cov=min_periods_cov, 

128 clip_at=clip_at, 

129 mean=mean, 

130 ) 

131 } 

132 for pair in pairs 

133 } 

134 

135 sigmas = _map_nested_dicts(results, lambda result: result.covariance) 

136 means = _map_nested_dicts(results, lambda result: result.mean) 

137 

138 # combination of covariance matrix valued time series 

139 return _CovarianceCombination(sigmas=sigmas, returns=returns, means=means)

146class _CovarianceCombination: 

147 def __init__(self, sigmas, returns, means=None): 

148 """ 

149 Computes the covariance combination of a set of covariance matrices 

150 

151 param sigmas: dictionary of covariance matrices {key: {time: sigma}} 

152 param returns: pandas DataFrame of returns 

153 param means: dictionary of means {key: {time: mu}}, optional 

154 

155 Note: sigmas[key][time] is a covariance matrix prediction for time+1 

156 """ 

157 # Assert Sigmas and means have same keys if means not None 

158 n = returns.shape[1] 

159 if means is not None: 

160 for key, sigma in sigmas.items(): 

161 # Assert sigmas and means have same keys 

162 assert sigma.keys() == means[key].keys(), "sigmas and means must have same keys" 

163 else: 

164 # Set means to zero if not provided 

165 means = {k: {time: np.zeros(n) for time in sigma.keys()} for k, sigma in sigmas.items()} 

166 

167 self.__means = means 

168 self.__sigmas = sigmas 

169 self.__returns = returns 

170 

171 # all those quantities don't depend on the window size 

172 self.__Ls = pd.DataFrame({k: _cholesky_precision(sigma) for k, sigma in self.sigmas.items()}) 

173 self.__Ls_shifted = self.__Ls.shift(1).dropna() 

174 self.__nus = pd.DataFrame({key: _nu(Ls, self.means[key]) for key, Ls in self.__Ls.items()}) 

175 self.__nus_shifted = self.__nus.shift(1).dropna()

203 def solve(self, window=None, times=None, **kwargs): 

204 """ 

205 The size of the window is crucial to specify the size of the parameters 

206 for the cvxpy problem. Hence those computations are not in the __init__ method 

207 

208 Solves the covariance combination problem at a time steps given in times 

209 

210 param window: number of previous time steps to use in the covariance 

211 combination problem 

212 param times: list of time steps to solve the problem at; if None, solve at all available time steps 

213 """ 

214 # If window is None, use all available data; cap window at length of data 

215 window = window or len(self.__Ls_shifted) 

216 window = min(window, len(self.__Ls_shifted)) 

217 

218 # Compute P matrix and its Cholesky factor 

219 Lts_at_r = pd.DataFrame( 

220 {key: _B_t_col(Ls, self.__nus_shifted[key], self.returns) for key, Ls in self.__Ls_shifted.items()} 

221 ) 

222 

223 # time steps to solve the problem at 

224 all_available_times = Lts_at_r.index 

225 

226 if times is None: 

227 times = Lts_at_r.index 

228 else: 

229 times = list(times) 

230 

231 # assert times are consecutive in all_available_times 

232 zero_index = all_available_times.get_loc(times[0]) 

233 assert list(all_available_times[zero_index : zero_index + len(times)]) == list(times) 

234 "times must be consecutive" 

235 

236 # add window - 1 previous time steps to times 

237 times = list(all_available_times[zero_index - window + 1 : zero_index]) + times 

238 

239 Bs = {time: np.column_stack(Lts_at_r.loc[time]) for time in times} 

240 prod_Bs = pd.Series({time: B.T @ B for time, B in Bs.items()}) 

241 # times = prod_Bs.index 

242 P = {times[i]: sum(prod_Bs.loc[times[i - window + 1] : times[i]]) for i in range(window - 1, len(times))} 

243 

244 P_chol = {time: np.linalg.cholesky(matrix) for time, matrix in P.items()} 

245 

246 # Compute A matrix 

247 Ls_diag = pd.DataFrame({k: _diag_part(L) for k, L in self.__Ls_shifted.items()}) 

248 

249 A = { 

250 times[i]: _A(Ls_diag.truncate(before=times[i - window + 1], after=times[i]), self.K) 

251 for i in range(window - 1, len(times)) 

252 } 

253 

254 problem = _CombinationProblem(keys=self.sigmas.keys(), n=len(self.assets), window=window) 

255 

256 problem._construct_problem() 

257 

258 for time, AA in A.items(): 

259 problem.A_param.value = AA 

260 problem.P_chol_param.value = P_chol[time] 

261 

262 try: 

263 yield self._solve(time=time, problem=problem, **kwargs) 

264 except cvx.SolverError: 

265 print(f"Solver did not converge at time {time}") 

266 yield None

268 def _solve(self, time, problem, **kwargs): 

269 """ 

270 Solves the covariance combination problem at a given time t 

271 """ 

272 problem.solve(**kwargs) 

273 

274 weights = problem.weights 

275 

276 # Get non-shifted L 

277 L = sum(self.__Ls.loc[time] * weights.values) # prediction for time+1 

278 nu = sum(self.__nus.loc[time] * weights.values) # prediction for time+1 

279 

280 mean = pd.Series(index=self.assets, data=np.linalg.inv(L.T) @ nu) 

281 sigma = pd.DataFrame(index=self.assets, columns=self.assets, data=np.linalg.inv(L @ L.T)) 

282 

283 mean = pd.Series(index=self.assets, data=np.linalg.inv(L.T) @ nu) 

284 sigma = pd.DataFrame(index=self.assets, columns=self.assets, data=np.linalg.inv(L @ L.T)) 

285 

286 return Result(time=time, mean=mean, covariance=sigma, weights=weights)
